from flask import Flask, render_template,request

app = Flask(__name__)


@app.route("/", methods=['GET', 'POST'])
def home():
    error = None
    if request.method== 'POST':
          name = request.form['name'].strip()
          
          if not name:
              error = "Name cannot be empty"
            
          else:  
            return f"Hello, {name}"
        
    return render_template("index.html",error=error)
